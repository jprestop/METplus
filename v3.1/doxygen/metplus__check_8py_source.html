12 def metplus_check_python_version(user_py, supported_py):
13  """!Test that the user's version of python is equal of higher than the
14  the supported version of python. Imported in each wrapper and master_metplus
15  to avoid confusing failures if the user's version is not current. Note:
16  SyntaxError from using f-strings (available in 3.6+) in earlier versions of
17  Python are output before the output from this function can be displayed.
18  Args:
19  @param user_py user's python version number, i.e. 3.8.1
20  @param supported_py currently supported python version number, i.e. 3.6.3
21  @returns True if version is at least supported, False if not
22  """
23  supported_list = supported_py.split('.')
24  user_list = user_py.split('.')
25 
26  for user, supported in zip(user_list, supported_list):
27  # if the same version is used, continue
28  if int(user) == int(supported):
29  continue
30 
31  # if a higher version is used, break out of the loop
32  if int(user) > int(supported):
33  break
34 
35  # a lower version is used - report and exit
36  print("ERROR: Must be using Python {} or higher ".format(supported_py)+
37  "with the required packages installed.")
38  print("You are using {}.".format(user_py))
39  print("See the METplus documentation for more information.")
40  return False
41 
42  return True

55 def evaluates_to_true(value):
56  """!Check if the value matches an expression that should be interpretted as False
57  Without this, environment variables that are set are interpretted as True no
58  matter what value they hold. This checks against common expressions like
59  false, no, off, and 0. This was leveraged from the get_bool logic in
60  produtil config. If the value does not match the False expressions, it is
61  considered to be True no matter what the value contains, i.e. mud
62  Args:
63  @param value value to check
64  @returns False if value matches a supported False expression, True otherwise
65  """
66  if re.match(r'(?i)\A(?:F|\.false\.|false|no|off|0)\Z', value):
67  return False
68 
69  return True

72  """! Check METPLUS_[DISABLE/ENABLE]_PLOT_WRAPPERS. If both are set it should error
73  and exit. Otherwise it should warn if DISABLE if used beacuse ENABLE should be
74  used instead.
75  @param environ dictionary containing environment variables that are set
76  @returns True if plot wrappers are enabled, False if they are disabled, and None
77  if both environment variables are set, which is not allowed
78  """
79  # if enable plot wrappers is set (to anything)
80  if environ.get('METPLUS_ENABLE_PLOT_WRAPPERS'):
81  # if disable plot wrappers is set, error and exit
82  if environ.get('METPLUS_DISABLE_PLOT_WRAPPERS'):
83  print("ERROR: Cannot set both METPLUS_ENABLE_PLOT_WRAPPERS and "
84  "METPLUS_DISABLE_PLOT_WRAPPERS environment variables.\n"
85  "Please unset METPLUS_DISABLE_PLOT_WRAPPERS and use "
86  "METPLUS_ENABLE_PLOT_WRAPPERS to enable plot wrappers.")
87  return None
88  else:
89  return evaluates_to_true(environ.get('METPLUS_ENABLE_PLOT_WRAPPERS'))
90  # if enable is not set but disable is, warn that disable is deprecated
