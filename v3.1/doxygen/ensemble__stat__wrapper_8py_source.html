250  def find_model_members(self, time_info):
251  """! Finds the model member files to compare
252  Args:
253  @param time_info dictionary containing timing information
254  @rtype string
255  @return Returns a list of the paths to the ensemble model files
256  """
257  model_dir = self.c_dict['FCST_INPUT_DIR']

259  fake_dir = '/ensemble/member/is/missing'
260 
261  # model_template is a list of 1 or more.
262  ens_members_path = []
263 
264  # get all files that exist
265  for ens_member_template in self.c_dict['FCST_INPUT_TEMPLATE']:
266  member_file = do_string_sub(ens_member_template,
267  **time_info)
268  expected_path = os.path.join(model_dir, member_file)
269 
270  # if wildcard expression, get all files that match
271  if '?' in expected_path or '*' in expected_path:
272  wildcard_files = sorted(glob.glob(expected_path))
273  self.logger.debug('Ensemble members file pattern: {}'
274  .format(expected_path))
275  self.logger.debug('{} members match file pattern'
276  .format(str(len(wildcard_files))))
277 
278  # add files to list of ensemble members
279  for wildcard_file in wildcard_files:
280  ens_members_path.append(wildcard_file)
281  else:
282  # otherwise check if file exists
283  expected_path = util.preprocess_file(expected_path,
284  self.c_dict['FCST_INPUT_DATATYPE'],
285  self.config)
286 
287  # if the file exists, add it to the list
288  if expected_path:
289  ens_members_path.append(expected_path)
290  else:
291  # add relative path to fake dir and add to list
292  ens_members_path.append(os.path.join(fake_dir, member_file))
293  self.logger.warning('Expected ensemble file {} not found'
294  .format(member_file))
295 
296  # if more files found than expected, error and exit
297  if len(ens_members_path) > self.c_dict['N_MEMBERS']:
298  msg = 'Found more files than expected! ' +\
299  'Found {} expected {}. '.format(len(ens_members_path),
300  self.c_dict['N_MEMBERS']) +\
301  'Adjust wildcard expression in [filename_templates] '+\
302  'FCST_ENSEMBLE_STAT_INPUT_TEMPLATE or adjust [config] '+\
303  'ENSEMBLE_STAT_N_MEMBERS. Files found: {}'.format(ens_members_path)
304  self.log_error(msg)
305  self.logger.error("Could not file files in {} for init {} f{} "
306  .format(model_dir, time_info['init_fmt'],
307  str(time_info['lead_hours'])))
308  return False
309  # if fewer files found than expected, warn and add fake files
310  elif len(ens_members_path) < self.c_dict['N_MEMBERS']:
311  msg = 'Found fewer files than expected. '+\
312  'Found {} expected {}.'.format(len(ens_members_path),
313  self.c_dict['N_MEMBERS'])
314  self.logger.warning(msg)
315  # add fake files to list to get correct number of files for ens_thresh
316  diff = self.c_dict['N_MEMBERS'] - len(ens_members_path)
317  self.logger.warning('Adding {} fake files to '.format(str(diff))+\
318  'ensure ens_thresh check is accurate')
319  for _ in range(0, diff, 1):
320  ens_members_path.append(fake_dir)
321 
322  # write file that contains list of ensemble files
323  list_filename = time_info['init_fmt'] + '_' + \
324  str(time_info['lead_hours']) + '_ensemble.txt'
325  return self.write_list_file(list_filename, ens_members_path)
