201  def getLowestForecastFile(self, valid_time, dtype, template):
202  """!Find the lowest forecast hour that corresponds to the
203  valid time
204  Args:
205  @param valid_time valid time to search
206  @param dtype data type (FCST or OBS) to get filename template
207  @rtype string
208  @return Path to file with the lowest forecast hour"""
209  out_file = None
210 
211  # search for file with lowest forecast, then loop up into you find a valid one
212  min_forecast = time_util.get_seconds_from_string(self.c_dict[dtype+'_MIN_FORECAST'], 'H')
213  max_forecast = time_util.get_seconds_from_string(self.c_dict[dtype+'_MAX_FORECAST'], 'H')
214  smallest_input_accum = min([lev['amount'] for lev in self.c_dict['ACCUM_DICT_LIST']])
215 
216  # if smallest input accumulation is greater than an hour, search hourly
217  if smallest_input_accum > 3600:
218  smallest_input_accum = 3600
219 
220  min_forecast_string = time_util.ti_get_lead_string(min_forecast)
221  max_forecast_string = time_util.ti_get_lead_string(max_forecast)
222  smallest_input_accum_string = time_util.ti_get_lead_string(smallest_input_accum, plural=False)
223  self.logger.debug(f"Looking for file with lowest forecast lead valid at {valid_time}"
224  f" between {min_forecast_string} and {max_forecast_string} using "
225  f"{smallest_input_accum_string} intervals")
226 
227  forecast_lead = min_forecast
228  while forecast_lead <= max_forecast:
229  input_dict = {}
230  input_dict['valid'] = valid_time
231  input_dict['lead_seconds'] = forecast_lead

234  fSts = do_string_sub(template,
235  **time_info)
236  search_file = os.path.join(self.input_dir,
237  fSts)

251  def get_daily_file(self, time_info, accum, data_src, file_template):
252  """!Pull accumulation out of file that contains a full day of data
253  Args:
254  @param time_info dictionary containing timing information
255  @param accum accumulation to extract from file
256  @param data_src type of data (FCST or OBS)
257  @param file_template filename template to search
258  @rtype bool
259  @return True if file was added to output list, False if not"""
260 
261  data_interval = self.c_dict[data_src + '_DATA_INTERVAL']
262  times_per_file = self.c_dict[data_src + '_TIMES_PER_FILE']
263  search_file = None
264  # loop from valid_time back to data interval * times per file
265  for i in range(0, times_per_file+1):
266  search_time = time_info['valid'] - datetime.timedelta(hours=(i * data_interval))
267  # check if file exists
268  dSts = do_string_sub(file_template,
269  valid=search_time,
270  custom=self.c_dict['CUSTOM_STRING'])
271  search_file = os.path.join(self.input_dir,
272  dSts)
273  search_file = util.preprocess_file(search_file,
274  self.c_dict[data_src+\
275  '_INPUT_DATATYPE'],
276  self.config)
277  if search_file is not None:
278  break
279 
280  if search_file is None:
281  return False
282 
283  diff = time_info['valid'] - search_time
284 
285  # Specifying integer division // Python 3,
286  # assuming that was the intent in Python 2.
287  lead = int((diff.days * 24) // (data_interval))
288  lead += int((diff).seconds // (data_interval*3600)) - 1
289  search_time_info = { 'valid' : search_time,
290  'custom': self.c_dict['CUSTOM_STRING']}
291 
292  # get name of input level item that matches the accumulation to extract from daily file
293  accum_seconds = time_util.get_seconds_from_string(accum, 'H')
294  accum_dict_list = self.c_dict['ACCUM_DICT_LIST']
295  fname = next((item['name'] for item in accum_dict_list if item['amount'] == accum_seconds), '-1')
296  # if accumulation was not found in levels dictionary list, error and return
297  if fname == '-1':
298  self.log_error(f'Accumulation {accum} was not specified in the {data_src}'
299  '_PCP_COMBINE_INPUT_ACCUMS list')
300  return False
301 
302  # if name was not set in the input levels list, use accumulation time in MET time format
303  if fname is None:
304  addon = time_util.time_string_to_met_time(accum, default_unit='S')
305  else:
306  fname = do_string_sub(fname, **search_time_info)
307  addon = "'name=\"" + fname + "\";"
308 
309  # if name is a python script, don't set level
310  if not util.is_python_script(fname):
311  addon += " level=\"(" + str(lead) + ",*,*)\";"
312 
313  addon += "'"
314 
315  self.add_input_file(search_file, addon)
316  return True

400  def get_accumulation(self, time_info, accum, data_src):
401  """!Find files to combine to build the desired accumulation
402  Args:
403  @param time_info dictionary containing time information
404  @param accum desired accumulation to build
405  @param data_src type of data (FCST or OBS)
406  @rtype bool
407  @return True if full set of files to build accumulation is found
408  """
409  in_template = self.c_dict[data_src+'_INPUT_TEMPLATE']
410 
411  if self.c_dict[data_src + '_IS_DAILY_FILE'] is True:
412  return self.get_daily_file(time_info, accum, data_src, in_template)
413 
414  search_time = time_info['valid']
415  # last time to search is the output accumulation subtracted from the
416  # valid time, then add back the smallest accumulation that is available
417  # in the input. This is done because data contains an accumulation from
418  # the file/field time backwards in time
419  # If building 6 hour accumulation from 1 hour accumulation files,
420  # last time to process is valid - 6 + 1
421  accum_relative = time_util.get_relativedelta(accum, 'H')
422  # using 1 hour for now
423  smallest_input_accum = min([lev['amount'] for lev in self.c_dict['ACCUM_DICT_LIST']])
424  if smallest_input_accum == 9999999:
425  smallest_input_accum = 3600
426 
427  last_time = time_info['valid'] -\
428  accum_relative +\
429  datetime.timedelta(seconds=smallest_input_accum)
430 
431  total_accum = time_util.ti_get_seconds_from_relativedelta(accum_relative,
432  time_info['valid'])
433 
434  # log the input and output accumulation information
435  search_accum_list = []
436  for lev in self.c_dict['ACCUM_DICT_LIST']:
437  if lev['template'] is not None:
438  search_accum_list.append(lev['template'])
439  else:
440  search_accum_list.append(time_util.ti_get_lead_string(lev['amount'], plural=False))
441 
442  search_accum_string = ' or '.join(search_accum_list)
443  self.logger.debug(f"Trying to build a {time_util.ti_get_lead_string(total_accum, plural=False)} accumulation using {search_accum_string} input data")
444 
445  # loop backwards in time until you have a full set of accum
446  while last_time <= search_time:
447  found = False
448 
449  if total_accum == 0:
450  break
451 
452  # look for biggest accum that fits search
453  for accum_dict in self.c_dict['ACCUM_DICT_LIST']:
454  if accum_dict['amount'] > total_accum and accum_dict['template'] is None:
455  continue
456 
457  search_file, lead = self.find_input_file(in_template, time_info['init'], search_time,
458  accum_dict['amount'], data_src)
459 
460  # if found a file, add it to input list with info
461  if search_file is not None:
462  # if template is used in accum, find value and apply bucket interval is set
463  if accum_dict['template'] is not None:
464  accum_amount = self.get_template_accum(accum_dict,
465  search_time,
466  lead,
467  data_src)
468  if accum_amount > total_accum:
469  self.logger.debug("Accumulation amount is bigger than remaining accumulation.")
470  continue
471  else:
472  accum_amount = accum_dict['amount']
473 
474  accum_met_time = time_util.time_string_to_met_time(accum_amount)
475  addon = self.get_addon(accum_dict, accum_met_time, search_time)
476  # add file to input list and step back in time to find more data
477  self.add_input_file(search_file, addon)
478  self.logger.debug(f"Adding input file: {search_file} with {addon}")
479  search_time = search_time - datetime.timedelta(seconds=accum_amount)
480  total_accum -= accum_amount
481  found = True
482  break
483 
484  # if we don't need any more accumulation, break out of loop and run
485  if not total_accum:
486  break
487 
488  # if we still need to find more accum but we couldn't find it, fail
489  if not found:
490  return False
491 
492  # fail if no files were found or if we didn't find
493  # the entire accumulation needed
494  if not self.infiles or total_accum:
495  return False
496 
497  return True

531  cmd += "-add "
