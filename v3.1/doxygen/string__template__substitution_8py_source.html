59 def get_tags(template):
60  """!Parse template and pull out all wildcard characters (* or ?) and all
61  tags, i.e. {init?fmt=%H}. Used to pull out information from a template that
62  contains wildcards and add that information when filled out another template
63  Returns a list of wildcards and tag names found, i.e. [ '*', 'init', 'lead']
64  """
65  i = 0
66  template_len = len(template)
67  tags = []
68  # loop through template looking for wildcard characters or tags: {init?fmt=%H}
69  while i < template_len:
70  if template[i] == TEMPLATE_IDENTIFIER_BEGIN:
71  end_i = template.find(TEMPLATE_IDENTIFIER_END, i)
72  tag = template[i+1:end_i]
73  identifier = tag.split('?')[0]
74  tags.append(identifier)
75  i = end_i
76  elif template[i] == '*' or template[i] == '?':
77  tags.append(template[i])
78 
79  i += 1
80  return tags

82 def format_one_time_item(item, time_str, unit):
83  """!Helper function for do_string_sub. Determine precision of time offset value and format
84  Args:
85  @param item format to substitute, i.e. 3M or H
86  @param time_str time value that precision will be applied, i.e. 60
87  @param unit currently being processed, i.e. M or H or S
88  Returns: Padded value or empty string if unit is not found in item
89  """
90  count = item.count(unit)
91  if count > 0:
92  rest = ''
93  # get precision from number (%3H)
94  res = re.match(r"^\.*(\d+)"+unit+"(.*)", item)
95  if res:
96  padding = int(res.group(1))
97  rest = res.group(2)
98  else:
99  padding = count
100  res = re.match("^"+unit+"+(.*)", item)
101  if res:
102  rest = res.group(1)
103  if unit != 's':
104  padding = max(2, count)
105 
106  # add formatted time
107  return str(time_str).zfill(padding)+rest
108 
109  # return empty string if no match
110  return ''

112 def format_hms(fmt, obj):
113  """!Helper function for do_string_sub. For time offset values, get hour, minute, and
114  second values to format as necessary
115  Args:
116  @param fmt format to substitute, i.e. %3H or %2M or %S
117  @param obj time value in seconds to format, i.e. 3600
118  Returns: Formatted time value
119  """
120  out_str = ''
121  fmt_split = fmt.split('%')[1:]
122  # split time into days, hours, mins, and secs
123  # time should be relative to the next highest unit if higher units are specified
124  # i.e. 90 minutes %M => 90, but %H%M => 0130
125  days = obj // 86400
126  hours = obj // 3600
127  minutes = obj // 60
128  seconds = obj
129 
130  # if days are specified, change hours, minutes, and seconds to relative
131  if True in ['d' in x for x in fmt_split]:
132  hours = (obj % 86400) // 3600
133  minutes = (obj % 3600) // 60
134  seconds = (obj % 3600) % 60
135 
136  # if hours are specified, change minutes and seconds to relative
137  if True in ['H' in x for x in fmt_split]:
138  minutes = (obj % 3600) // 60
139  seconds = (obj % 3600) % 60
140 
141  # if minutes are specified, change seconds to relative
142  if True in ['M' in x for x in fmt_split]:
143  seconds = (obj % 3600) % 60
144 
145  # parse format
146  for item in fmt_split:
147  out_str += format_one_time_item(item, hours, 'H')
148  out_str += format_one_time_item(item, minutes, 'M')
149  out_str += format_one_time_item(item, seconds, 'S')
150  out_str += format_one_time_item(item, obj, 's')
151  out_str += format_one_time_item(item, days, 'd')
152 
153  return out_str

234 def round_time_down(obj, truncate_seconds):
235  """!If template value needs to be truncated, round the value down
236  to the given truncate interval"""
237  if truncate_seconds == 0:
238  return obj
239 
240  trunc = truncate_seconds
241  seconds = (obj.replace(tzinfo=None) - obj.min).seconds

298 def do_string_sub(tmpl, skip_missing_tags=False, **kwargs):
299  """
300  log - log object
301  tmpl_str - template string to populate
302  kwargs - dictionary containing values for each template key
303 
304  This class provides functionality for substituting values for
305  string templates.
306 
307  Possible keys for vals:
308  init - datetime object
309  valid - datetime object
310  lead - must be in seconds
311  level - must be in seconds
312  model - the name of the model
313  domain - the domain number (01, 02, etc.) read in as a string
314  cycle - the cycle hour in HH format (00, 03, 06, etc.)
315  offset_hour - the offset hour in HH format to add
316  to the init + cycl time:
317  (YYYYMMDD + hh) + offset
318  Indicate a negative offset by using a '-' sign
319  in the
320  date - datetime object
321  region - the two-character (upper or lower case) region/basin designation
322  cyclone - a two-digit annual cyclone number (if ATCF_by_pairs) or
323  four-digit cyclone number (leading zeros)
324  misc - any string
325 
326  Populates the specified template with information from the
327  kwargs dictionary. The template structure is comprised of
328  a fixed string populated with template place-holders inside curly
329  braces, for example {tmpl_str}. The tmpl_str must be present as
330  a key in the kwargs dictionary, and the value will replace the
331  {tmpl_str} in the returned string.
332 
333  In some cases, the template keys can have parameters containing
334  formatting information. The format of the template in this case
335  is {tmpl_str?parm=val}. The supported parameters are:
336 
337  init, valid:
338  fmt - specifies a strftime format for the date/time
339  e.g. %Y%m%d%H%M%S, %Y%m%d%H
340 
341  lead, level:
342  fmt - specifies an amount of time in [H]HH[MMSS] format
343  e.g. %HH, %HHH, %HH%MMSS, %HHH%MMSS
344 
345  cycle, negative_offset, positive_offset:
346  fmt - specifies the cycle and offset hours in HH format. H and
347  HHH format are supported, to anticipate any changes
348  in prepbufr data.
349 
350  The following were created to support processing of
351  tropical cyclone data:
352  cyclone:
353  fmt - specifies the annual cyclone number as a string.
354 
355  region:
356  fmt - a string that specifies the region/basin of cyclone.
357  For ATCF_by_pairs formatted data, this is a 2-character
358  designation:
359  AL|WP|CP|EP|SH|IO|LS lower case designations are
360  observed in filenames.
361  date:
362  fmt - a string representation of the date format for a
363  subdirectory in which track data resides, or the
364  string representation of the date format in a file.
365  Expected formats: YYYY, YYYYMM, YYYYMMDD, and YYYYMMDDhh
366  misc:
367  fmt -a string that represents any other miscellaneous feature
368  of the track data, such as experiment name or some other descriptor
369  """

372  # following + matches 1 or more occurrence of preceding expression.
373  # The ? after the .+ makes it a lazy match so that it stops

491 def populate_match_dict(template, filepath, logger=None):
492  """! Use template to extract time information from filepath, add each value to a dictionary.
493  Populates a dictionary with keys that contain tag name + time type, i.e. init+Y,
494  valid+M, or lead+S, with string values containing the number extracted from the filepath.
495  Also determines the shift amount for valid time if it was found, i.e.
496  {valid?fmt=%Y%m%d?shift=-30}. Valid shift will be 0 if no shift.
497  Note: valid time values will not have the shift applied.
498  Args:
499  @param template filename template to use to find time information, i.e.
500  file.{valid?fmt=%Y%m%d}.ext
501  @param filepath path to examine, i.e. file.20190201.ext
502  @returns tuple of match dictionary and valid shift value if success, i.e.
503  ({'init+Y': '2019'}, -30)
504  Returns (None, None) if could not extract time info
505  """
506  # get the text before any tags, between tags, and after any tags
507  match = re.match(r'([^{]*)({.*})([^}]*)', template)
508  if not match:
509  # if there were no tags, we can't extract time info, so return None, None
510  if logger:
511  logger.debug("No tags found (1)")
512  return None, None
513 
514  # tags to process, including text in between tags
515  all_tags = match.group(2)
516 
517  # check if text before and after tags matches template and strip off from file path
518  filepath = check_pre_text(filepath, match.group(1), logger)
519  filepath = check_post_text(filepath, match.group(3), logger)
520  if filepath is None:
521  return None, None
522 
523  return process_match_tags(all_tags, filepath, logger)

651  # extract string that corresponds to format

675 def get_fmt_info(fmt, filepath, match_dict, identifier):
676  """!Helper function for parse_template. Reads format information from tag and
677  populates dictionary with extracted values.
678  Args:
679  @param fmt formatting values from template tag, i.e. %Y%m%d
680  @param filepath rest of text from filename that can be parsed
681  @param match_dict dictionary of extracted information. Key is made up
682  of the identifier and the format tag, i.e. init_H or valid_M.
683  Value is the extracted information, i.e. 19870201
684  @param identifier tag name, i.e. 'init' or 'lead'
685  @returns Number of characters processed from the filename if success,
686  -1 if failed to parse all format items in template tag"""
687  length = 0
688  # find all items that start with %, i.e. %Y or %3H, %10S, or %.2d
689  # group 1 is optional number, i.e. 3 in %3H or 2 in %.2d
690  # group 2 is letters with extra characters at the end,
691  # i.e. HH in %HH or S: in %10S:
692  match_list = re.findall(r'%\.?(\d*)([^%]+)', fmt)
693  for match in match_list:
694  # letter identifier for time - may be duplicate letters
695  # i.e. H in %3H or HHH in %HHH or H: in %H: (can have trailing characters)
696  time_letter = match[1][0]
697 
698  # optional number for specifying zero padding, i.e. 3 in %3H
699  time_number = match[0]
700 
701  # if first letter is not found in time identifier length dictionary
702  if time_letter not in LENGTH_DICT.keys():
703  return None
704 
705  # get length of time type
706  new_len = LENGTH_DICT.get(time_letter)
707 
708  # find how many times the first letter is found in the time letters
709  # i.e. HHH: should be 3
710  match_len = re.match(r'([' + time_letter + ']+)(.*)', match[1])
711  if not match_len:
712  return None
713 
714  # if there are multiple letters, update length if it is more than
715  # the standard length for that time type
716  time_letter_count = len(match_len.group(1))
717  extra_len = len(match_len.group(2))
718  if time_letter_count > 1:
719  # if multiple letters and a number is specifed, i.e. 3HH, then fail
720  if time_number:
721  return None
722 
723  new_len = time_letter_count
724 
725  # if number was provided and it is different than the
726  # standard length use that number instead
727  elif time_number and int(time_number) != new_len:
728  new_len = int(time_number)
729 
730  # if lead or level hours are found
731  if (match[1] == 'H' and
732  (identifier in ('lead', 'level'))):
733  # look forward until non-digit is found
734  new_len = 0
735  while new_len < len(filepath) and filepath[new_len].isdigit():
736  new_len += 1
737 
738  # add the length specified plus any additional characters
739  length += new_len + extra_len
740 
741  if not add_to_dict(identifier + '+' + time_letter,
742  match_dict,
743  filepath,
744  new_len):
745  return None
746 
747  filepath = filepath[new_len+extra_len:]
748 
749  return length
