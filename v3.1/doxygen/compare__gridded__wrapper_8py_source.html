98  def check_probabilistic_settings(self):
99  """!If dataset is probabilistic, check if *_PROB_IN_GRIB_PDS or INPUT_DATATYPE
100  are set. If not enough information is set, report an error and set isOK to False"""
101  for dtype in ['FCST', 'OBS']:
102  if self.c_dict[f'{dtype}_IS_PROB']:
103  # if the data type is NetCDF, then we know how to
104  # format the probabilistic fields
105  if self.c_dict[f'{dtype}_INPUT_DATATYPE'] == 'NETCDF':
106  continue
107 
108  # if the data is grib, the user must specify if the data is in
109  # the GRIB PDS or not
110  if self.c_dict[f'{dtype}_PROB_IN_GRIB_PDS'] == '':
111  self.log_error(f"If {dtype}_IS_PROB is True, you must set {dtype}_PROB_IN"
112  "_GRIB_PDS unless the forecast datatype is set to NetCDF")
113  self.isOK = False
