237  stat_analysis_exe = os.path.join(self.config.getdir('MET_BIN_DIR'),
238  'stat_analysis')
239  p = subprocess.Popen([stat_analysis_exe, "--version"],
240  stdout=subprocess.PIPE)
241  out, err = p.communicate()
242  out = out.decode(encoding='utf-8', errors='strict')
243  for line in out.split('\n'):
244  if 'MET Version:' in line:
245  met_verison_line = line
246  met_version_str = (
247  met_verison_line.partition('MET Version:')[2].split('V')[1]
248  )
249  if len(met_version_str) == 3:
