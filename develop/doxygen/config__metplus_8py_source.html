154 def parse_launch_args(args, logger, base_confs=METPLUS_BASE_CONFS):
155  """!Parsed arguments to scripts that launch the METplus system.
156 
157  This is the argument parser for the config_metplus.py
158  script. It parses the arguments (in args).
159  If something goes wrong, this function calls
160  sys.exit(1) or sys.exit(2).
161 
162  Options:
163  * section.variable=value --- set this value in this section, no matter what
164  * /path/to/file.conf --- read this conf file after the default conf files.
165 
166  Later conf files override earlier ones. The conf files read in
167  are:
168  * metplus.conf
169 
170  @param args the script arguments, after script-specific ones are removed
171  @param logger a logging.Logger for log messages"""
172 
173  parm = os.path.realpath(PARM_BASE)
174 
175  # Files in this list, that don't exist or are empty,
176  # will be silently ignored.
177  # infiles=[ os.path.join(parm, 'metplus.conf'),
178  # os.path.join(parm, 'metplus.override.conf')
179  # ]
180  infiles = list()
181  for base_conf in base_confs:
182  infiles.append(base_conf)
183 
184  moreopt = collections.defaultdict(dict)
185 
186  if args is None:
187  return (parm, infiles, moreopt)
188 
189  # Now look for any option and conf file arguments:
190  bad = False
191  for iarg in range(len(args)):
192  m = re.match(r'''(?x)
193  (?P<section>[a-zA-Z][a-zA-Z0-9_]*)
194  \.(?P<option>[^=]+)
195  =(?P<value>.*)$''', args[iarg])
196  if m:
197  logger.info('Set [%s] %s = %s' % (
198  m.group('section'), m.group('option'),
199  repr(m.group('value'))))
200  moreopt[m.group('section')][m.group('option')] = m.group('value')
201  elif os.path.exists(args[iarg]):
202  infiles.append(args[iarg])
203  elif os.path.exists(os.path.join(parm, args[iarg])):
204  infiles.append(os.path.join(parm, args[iarg]))
205  else:
206  bad = True
207  # if OS separator character (/ or \) if in argument, assume it
208  # is supposed to be a path but the file is not found
209  if os.sep in args[iarg]:
210  logger.error(f"Configuration file not found: {args[iarg]}")
211  else:
212  logger.error(f"Invalid argument: {args[iarg]}")
213  if bad:
214  sys.exit(2)

352  log_timestamp_template = config.getstr('config', 'LOG_TIMESTAMP_TEMPLATE', '')
353  if log_timestamp_template:
354  # Note: strftime appears to handle if log_timestamp_template
355  # is a string ie. 'blah' and not a valid set of % directives %Y%m%d,
356  # it does return the string 'blah', instead of crashing.
357  # However, I'm still going to test for a valid % directive and

362  # directive in the conf file.

364  # case insensitive %(AnyAlphabeticCharacter), ie. %Y%m ...
365  # The basic pattern is (%+[a-z])+ , %+ allows for 1 or more

626  Args:

630  @param count: Counter used to stop recursion to prevent infinite

633  """
634  count = count + 1
635  if count >= 10:
636  return ''
637

639  # look in the [config] section for the variable
