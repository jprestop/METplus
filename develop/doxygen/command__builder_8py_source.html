258  def get_env_copy(self, var_list=None):
259  """!Print list of environment variables that can be easily
260  copied into terminal
261  """
262  out = ""
263  if not var_list:
264  var_list = self.env_list
265 
266  if 'user_env_vars' in self.config.sections():
267  for user_var in self.config.keys('user_env_vars'):
268  var_list.add(user_var)
269 
270  shell = self.config.getstr('config', 'USER_SHELL', 'bash').lower()
271  for var in sorted(var_list):
272  if shell == 'csh':
273  # TODO: Complex environment variables that have special characters
274  # like { or } will not be copyable in csh until modifications are
275  # made to the formatting of the setenv calls
276  clean_env = self.env[var].replace('"', '"\\""')
277  line = 'setenv ' + var + ' "' + clean_env + '"'
278  else:
279  # insert escape characters to allow export command to be copyable
280  clean_env = self.env[var].replace('"', r'\"').replace(r'\\"', r'\\\"')
281  line = 'export ' + var + '="' + clean_env + '"'
282 
283  out += line + '; '

299  """!Handles config variables that have fcst/obs versions or a generic

309  if has_fcst and has_obs:
310  fcst_conf = self.config.getstr(sec, fcst_name)
311  obs_conf = self.config.getstr(sec, obs_name)
312  if has_gen:
313  self.logger.warning('Ignoring conf {} and using {} and {}'

315  return fcst_conf, obs_conf
316 
317  # if one but not the other is set, error and exit
318  if has_fcst and not has_obs:
319  self.log_error('Cannot use {} without {}'.format(fcst_name, obs_name))
320  return None, None
321 
322  if has_obs and not has_fcst:
323  self.log_error('Cannot use {} without {}'.format(obs_name, fcst_name))
324  return None, None
325 
326  # if generic conf is set, use for both
327  if has_gen:
328  gen_conf = self.config.getstr(sec, gen_name)

334  obs_name)

473  def find_exact_file(self, level, data_type, time_info, mandatory=True, return_list=False, allow_dir=False):
474  input_template = self.c_dict.get(f'{data_type}INPUT_TEMPLATE', '')
475  data_dir = self.c_dict.get(f'{data_type}INPUT_DIR', '')
476 
477  if not input_template:
478  self.log_error(f"Could not find any {data_type}INPUT files because no template was specified")
479  return None
480 
481  check_file_list = []
482  found_file_list = []
483 
484  # check if there is a list of files provided in the template
485  # process each template in the list (or single template)
486  template_list = util.getlist(input_template)
487 
488  # return None if a list is provided for a wrapper that doesn't allow
489  # multiple files to be processed
490  if len(template_list) > 1 and not self.c_dict.get('ALLOW_MULTIPLE_FILES', False):
491  self.log_error("List of templates specified for a wrapper that "
492  "does not allow multiple files to be provided.")
493  return None
494 
495  for template in template_list:
496  # perform string substitution
497  filename = do_string_sub(template,
498  level=level,
499  **time_info)
500 
501  # build full path with data directory and filename
502  full_path = os.path.join(data_dir, filename)
503 
504  if os.path.sep not in full_path:
505  self.logger.debug(f"{full_path} is not a file path. Returning that string.")
506  return full_path
507 
508  self.logger.debug(f"Looking for {data_type}INPUT file {full_path}")
509 
510  # if wildcard expression, get all files that match
511  if '?' in full_path or '*' in full_path:
512 
513  wildcard_files = sorted(glob.glob(full_path))
514  self.logger.debug(f'Wildcard file pattern: {full_path}')
515  self.logger.debug(f'{str(len(wildcard_files))} files match pattern')
516 
517  # add files to list of files
518  for wildcard_file in wildcard_files:
519  check_file_list.append(wildcard_file)
520  else:
521  # add single file to list
522  check_file_list.append(full_path)
523 
524  # if multiple files are not supported by the wrapper and multiple files are found, error and exit
525  # this will allow a wildcard to be used to find a single file. Previously a wildcard would produce
526  # an error if only 1 file is allowed.
527  if not self.c_dict.get('ALLOW_MULTIPLE_FILES', False) and len(check_file_list) > 1:
528  self.log_error("Multiple files found when wrapper does not support multiple files.")
529  return None
530 
531  # return None if no files were found
532  if not check_file_list:
533  msg = f"Could not find any {data_type}INPUT files"
534  if mandatory:
535  self.log_error(msg)
536  else:
537  self.logger.warning(msg)
538 
539  return None
540 
541  for file_path in check_file_list:
542  # check if file exists
543  input_data_type = self.c_dict.get(data_type + 'INPUT_DATATYPE', '')
544  processed_path = util.preprocess_file(file_path,
545  input_data_type,
546  self.config,
547  allow_dir=allow_dir)
548 
549  # report error if file path could not be found
550  if not processed_path:
551  msg = f"Could not find {data_type}INPUT file {file_path} using template {template}"
552  if mandatory:
553  self.log_error(msg)
554  else:
555  self.logger.warning(msg)
556 
557  return None
558 
559  if os.path.isdir(processed_path):
560  self.logger.debug(f"Found directory: {processed_path}")
561  else:
562  self.logger.debug(f"Found file: {processed_path}")
563  found_file_list.append(processed_path)
564 
565  # if only one item found and return_list is False, return single item
566  if len(found_file_list) == 1 and not return_list:
567  return found_file_list[0]
568 
569  return found_file_list

571  def find_file_in_window(self, level, data_type, time_info, mandatory=True, return_list=False):
572  template = self.c_dict[f'{data_type}INPUT_TEMPLATE']
573  data_dir = self.c_dict[f'{data_type}INPUT_DIR']
574 
575  # convert valid_time to unix time
576  valid_time = time_info['valid_fmt']
577  valid_seconds = int(datetime.strptime(valid_time, "%Y%m%d%H%M%S").strftime("%s"))
578  # get time of each file, compare to valid time, save best within range
579  closest_files = []
580  closest_time = 9999999
581 
582  # get range of times that will be considered
583  valid_range_lower = self.c_dict.get(data_type + 'FILE_WINDOW_BEGIN', 0)
584  valid_range_upper = self.c_dict.get(data_type + 'FILE_WINDOW_END', 0)
585  lower_limit = int(datetime.strptime(util.shift_time_seconds(valid_time, valid_range_lower),
586  "%Y%m%d%H%M%S").strftime("%s"))
587  upper_limit = int(datetime.strptime(util.shift_time_seconds(valid_time, valid_range_upper),
588  "%Y%m%d%H%M%S").strftime("%s"))
589 
590  msg = f"Looking for {data_type}INPUT files under {data_dir} within range " +\
591  f"[{valid_range_lower},{valid_range_upper}] using template {template}"
592  self.logger.debug(msg)
593 
594  if not data_dir:
595  self.log_error('Must set INPUT_DIR if looking for files within a time window')
596  return None
597 
598  # step through all files under input directory in sorted order
599  for dirpath, _, all_files in os.walk(data_dir):
600  for filename in sorted(all_files):
601  fullpath = os.path.join(dirpath, filename)
602 
603  # remove input data directory to get relative path
604  rel_path = fullpath.replace(f'{data_dir}/', "")
605  # extract time information from relative path using template
606  file_time_info = util.get_time_from_file(rel_path, template, self.logger)
607  if file_time_info is None:
608  continue
609 
610  # get valid time and check if it is within the time range
611  file_valid_time = file_time_info['valid'].strftime("%Y%m%d%H%M%S")
612  # skip if could not extract valid time
