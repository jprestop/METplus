148  def reformat_grid_id(self, grid_id):
149  """!Reformat the grid id (MASK_GRID value in the configuration
150  file) if it starts with G. Looks for G<n> where n is a
151  digit 0-999 and zero pads the value. i.e. G7 becomes G007,
152  G13 becomes G013, etc.
153  Args:
154  @param grid_id identifier of grid
155  Returns:
156  @return reformatted grid id if valid, None if not
157  """
158  # If grid ID does not start with G, return it
159  if not grid_id.startswith('G'):
160  return grid_id
161 
162  # look for G<n> where n is a digit 0-999
163  match = re.match(r'G([0-9]{1,3})$', grid_id)
164 
165  # pad with zeros if found
166  if match:
167  number = match.group(1)
168  return 'G' + number.zfill(3)
169 
170  # Unexpected format
171  self.log_error('Grid id in unexpected format of Gn or ' +
172  'Gnn, please check again. Exiting...')
173  return None

293  def get_command(self):
294  """! Builds the command to run the MET application
295  @rtype string
296  @return Returns a MET command with arguments that you can run
297  """
298  if self.app_path is None:
299  self.log_error('No app path specified. You must use a subclass')
300  return None
301 
302  cmd = '{} -v {} '.format(self.app_path, self.c_dict['VERBOSITY'])
303 
304  for a in self.args:
305  cmd += a + " "
306 
307  if len(self.infiles) == 0:
308  self.log_error("No input filenames specified")
309  return None
310 
311  # if multiple input files, add first now, then add rest with
312  # -pbfile argument
313  cmd += self.infiles[0] + " "
314 
315  if self.outfile == "":
316  self.log_error("No output filename specified")
317  return None
318 
319  if self.outdir == "":
320  self.log_error("No output directory specified")
321  return None
322 
323  out_path = os.path.join(self.outdir, self.outfile)
324 
325  # create outdir (including subdir in outfile) if it doesn't exist
326  if not os.path.exists(os.path.dirname(out_path)):
327  os.makedirs(os.path.dirname(out_path))
328 
329  cmd += out_path + ' '
330 
331  cmd += self.c_dict['CONFIG_FILE'] + ' '
332 
333  if len(self.infiles) > 1:
334  for f in self.infiles[1:]:
335  cmd += '-pbfile ' + f + ' '
336
