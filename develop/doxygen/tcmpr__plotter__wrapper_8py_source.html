143

145 
146  Returns:

148  """
149 
150  self.logger.debug("TCMPR input " + self.input_data)
151  self.logger.debug("TCMPR config file " +
152  self.plot_config_file)
153  self.logger.debug("output " + self.output_base_dir)
154 
155  # Create a dictionary of all the "optional" options and flags.
156  cmds_dict = self.retrieve_optionals()
157 
158  # Create the TCMPR output base directory, where the final plots
159  # will be saved.
160  util.mkdir_p(self.output_base_dir)
161 
162  # If input data is a file, create a single command and invoke R script.
163  if os.path.isfile(self.input_data):
164  self.logger.debug("Currently plotting " + self.input_data)
165  cmds_dict[' -lookin '] = self.input_data
166 
167  # Special treatment of the "optional" output_base_dir option
168  # because we are supporting the plotting of multiple tcst files
169  # in a directory.
170  if self.output_base_dir:
171  # dated_output_dir = self.create_output_subdir(self.input_data)
172  cmds_dict[' -outdir '] = self.output_base_dir
173 
174  # Generate the list, where the -args are separated by their
175  # values.
176  full_cmd_list = ['Rscript' + self.tcmpr_script]
177  for key, value in cmds_dict.items():
178  full_cmd_list.append(key)
179  full_cmd_list.append(value)
180 
181  # Separate the 'Rscript' portion from the args, to conform to
182  # produtil's exe syntax.
183  cmd = exe(full_cmd_list[0])[full_cmd_list[1:]] > '/dev/null'
184  self.logger.debug("Command run " +
185  cmd.to_shell())
186  self.logger.info("Generating requested plots for " +
187  self.input_data)
188  # pylint:disable=unnecessary-pass
189  # If a tc file is empty, continue to the next, thus the pass
190  # isn't unnecessary.
191  try:
192  checkrun(cmd)
193  except produtil.run.ExitStatusException as ese:
194  self.logger.warn("plot_tcmpr.R returned non-zero"
195  " exit status, "
196  "tcst file may be missing data, "
197  "continuing: " + repr(ese))
198 
199  # If the input data is a directory, create a list of all the
200  # files in the directory and invoke the R script for this list
201  # of files.
202  elif os.path.isdir(self.input_data):
203  self.logger.debug("plot all files in directory " +
204  self.input_data)
205  cmds_dict = self.retrieve_optionals()
206  all_tcst_files_list = util.get_files(self.input_data, ".*.tcst",
207  self.logger)
208  all_tcst_files = ' '.join(all_tcst_files_list)
209  self.logger.debug("num of files " + str(len(all_tcst_files)))
210  # Append the mandatory -lookin option to the base command.
211  cmds_dict['-lookin'] = all_tcst_files
212  if self.output_base_dir:
213  cmds_dict['-outdir'] = self.output_base_dir
214  self.logger.debug("Creating dated output dir " +
215  self.output_base_dir)
216 
217  # Create the full_cmd_list from the keys and values of the
218  # cmds_dict and then form one command list.
219  full_cmd_list = list()
220  full_cmd_list.append("Rscript")
221  full_cmd_list.append(self.tcmpr_script)
222  for key, value in cmds_dict.items():
223  full_cmd_list.append(key)
224  if key == '-lookin':
225  # treat the list of dirs in -lookin differently,
226  # append each individual directory to replicate original
227  # implementation's behavior of splitting the commands
228  # by whitespace and assigning each command to an item
229  # in a list.
230  for tcst_file in all_tcst_files_list:
231  full_cmd_list.append(tcst_file)
232  elif key == '-plot':
233  # plot types list is also appended as a single string,
234  # delimited by ','.
235  full_cmd_list.append(','.join(value))
236  elif key == '-dep':
237  # dependant variables list items are appended
238  # as one string. Convert list into a string delimited
239  # by ','.
240  full_cmd_list.append(','.join(value))
241 
242  else:
243  full_cmd_list.append(value)
244 
245  # Separate the 'Rscript' portion from the args, to conform to
246  # produtil's exe syntax.
247  cmd = exe(full_cmd_list[0])[full_cmd_list[1:]] > '/dev/null'
248 
249  # This can be a very long command if the user has
250  # indicated a directory. Only log this if necessary.
251  # self.logger.debug("DEBUG: Command run " + cmd.to_shell())
252  # cmd_str = ' '.join(full_cmd_list)
253  # cmd_list = 'Rscript ' + cmd_str
254  # self.logger.debug('TCMPR Command run: ' + cmd_str)
255 
256  # Now run the command via produtil
257  try:
258  checkrun(cmd)
259  except produtil.run.ExitStatusException as ese:
260  # If the tcst file is empty (with the exception of the
261  # header), or there is some other problem, then
262  # plot_tcmpr.R will return with a non-zero exit status of 1
263  self.log_error("plot_tcmpr.R returned non-zero"
264  " exit status, tcst file may be missing"
265  " data... continuing: " + str(ese))
266  sys.exit(1)
267  else:
268  self.log_error("Expected input is neither a file nor directory,"
269  "exiting...")
270  sys.exit(1)
271 
272  self.logger.info("Plotting complete")
