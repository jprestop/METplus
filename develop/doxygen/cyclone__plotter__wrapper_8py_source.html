284  def extract_date_and_time_from_init(init_time_str):
285  """ Extract and return the YYYYMMDD portion and the
286  hh portion from the init time taken from the .tcst file.
287  """
288  match_ymd = re.match(r'([0-9]{8}).*', init_time_str)
289  match_hh = re.match(r'[0-9]{8}_([0-9]{2,3})[0-9]{4}', init_time_str)
290  # pylint:disable=no-else-return
291  # Explicitly return None if no match is found
292  if match_ymd and match_hh:
293  return match_ymd.group(1), match_hh.group(1)
294  else:
295  return None
