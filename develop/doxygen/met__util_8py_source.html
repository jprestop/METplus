145 def run_metplus(config, process_list):
146  total_errors = 0
147 
148  try:
149  processes = []
150  for item in process_list:
151  try:
152  logger = config.log(item)
153  package_name = 'metplus.wrappers.' + camel_to_underscore(item) + '_wrapper'
154  module = import_module(package_name)
155  command_builder = getattr(module,
156  item + "Wrapper")(config)
157 
158  # if Usage specified in PROCESS_LIST, print usage and exit
159  if item == 'Usage':
160  command_builder.run_all_times()
161  return 0
162  except AttributeError:
163  raise NameError("There was a problem loading %s wrapper." % item)
164 
165  processes.append(command_builder)
166 
167  # check if all processes initialized correctly
168  allOK = True
169  for process in processes:
170  if not process.isOK:
171  allOK = False
172  class_name = process.__class__.__name__.replace('Wrapper', '')
173  logger.error("{} was not initialized properly".format(class_name))
174 
175  # exit if any wrappers did not initialized properly
176  if not allOK:
177  logger.info("Refer to ERROR messages above to resolve issues.")
178  return 1
179 
180  loop_order = config.getstr('config', 'LOOP_ORDER', '')
181 
182  if loop_order == "processes":
183  for process in processes:
184  process.run_all_times()
185 
186  elif loop_order == "times":
187  loop_over_times_and_call(config, processes)
188 
189  else:
190  logger.error("Invalid LOOP_ORDER defined. " + \
191  "Options are processes, times")
192  return 1
193 
194  # compute total number of errors that occurred and output results
195  for process in processes:
196  if process.errors != 0:
197  process_name = process.__class__.__name__.replace('Wrapper', '')
198  error_msg = '{} had {} error'.format(process_name, process.errors)
199  if process.errors > 1:
200  error_msg += 's'
201  error_msg += '.'
202  logger.error(error_msg)
203  total_errors += process.errors
204 
205  return total_errors
206  except:
207  logger.exception("Fatal error occurred")
208  logger.info(f"Check the log file for more information: {config.getstr('config', 'LOG_METPLUS')}")
209  return 1

211 def post_run_cleanup(config, app_name, total_errors):
212  logger = config.logger
213  # scrub staging directory if requested
214  if config.getbool('config', 'SCRUB_STAGING_DIR', False) and\
215  os.path.exists(config.getdir('STAGING_DIR')):
216  staging_dir = config.getdir('STAGING_DIR')
217  logger.info("Scrubbing staging dir: %s", staging_dir)
218  shutil.rmtree(staging_dir)
219 
220  # rewrite final conf so it contains all of the default values used
221  write_final_conf(config, logger)
222 
223  # compute time it took to run
224  start_clock_time = datetime.datetime.strptime(config.getstr('config', 'CLOCK_TIME'),
225  '%Y%m%d%H%M%S')
226  end_clock_time = datetime.datetime.now()
227  total_run_time = end_clock_time - start_clock_time
228  logger.debug(f"{app_name} took {total_run_time} to run.")
229 
230  log_message = f"Check the log file for more information: {config.getstr('config', 'LOG_METPLUS')}"
231  if total_errors == 0:
232  logger.info(log_message)
233  logger.info(f'{app_name} has successfully finished running.')
234  else:
235  error_msg = f"{app_name} has finished running but had {total_errors} error"
236  if total_errors > 1:
237  error_msg += 's'
238  error_msg += '.'
239  logger.error(error_msg)
240  logger.info(log_message)
241  sys.exit(1)
